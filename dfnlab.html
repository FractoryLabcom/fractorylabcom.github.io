DFNLab¶

Introduction¶

DFNLab is a collection of modules to deal with Discrete Fracture Networks. These modules are written in C++ and binded in Python using pybind11. Mathematical operations are performed with Eigen3. Some visualizations are performed with VTK. Mesh generation is performed using BAMG.

Contact: fractory@itasca.fr

def plot_dfn(fnet,plotDFN=False, plotInter=False):
    fig = pl.figure()
    ax = a3.Axes3D(fig)
    ax.axis('equal')
    scube = fnet.sysSizeCube()
    ax.set_xlim3d(-scube[0]/2, scube[0]/2)
    ax.set_ylim3d(-scube[1]/2, scube[1]/2)
    ax.set_zlim3d(-scube[2]/2, scube[2]/2)
    smax = max(scube[0],scube[1],scube[2])/4
    if plotInter:
        for ii in fnet.iterIntersectionsFractures():
           ls = np.hstack([ii.getExt1(), ii.getExt2()]).copy()
           ls = ls.reshape((-1,2,3))
           seg = a3.art3d.Line3DCollection(ls,linewidth=2)
           seg.set_color('k')
           ax.add_collection3d(seg)
    jet = pl.get_cmap('jet') 
    cNorm = colors.LogNorm(0.01,smax)
    scalarMap = cmx.ScalarMappable(norm=cNorm, cmap=jet)
    cNorm = colors.Normalize(0,fnet.nbFractures())
    scalarMap = cmx.ScalarMappable(norm=cNorm, cmap=jet)
    a=[]
    if plotDFN:
        for fi in fnet:
            a.append(fi.getIndex())
            a.append(fi.area())
        scalarMap.set_array(a)
        for fi in fnet:
            vlist = fi.vertList()
            tri = a3.art3d.Poly3DCollection([vlist],linewidth=0.5, alpha=0.5)
            farea = fi.area()
            a.append(farea)
            colorVal = scalarMap.to_rgba(math.log(farea))
            colorVal = scalarMap.to_rgba(fi.getIndex())
            tri.set_color(colorVal)
            tri.set_edgecolor('k')
            ax.add_collection3d(tri)
        
    vticks=np.logspace(math.log10(0.01),math.log10(smax),8)
    vticks = np.linspace(0,fnet.nbFractures(), 9)
    fig.colorbar(scalarMap,shrink=0.9,ticks=vticks,format= '%.2f')
    pl.show()

 
def plot_stereonet(dip,strike,weight):
        fig, ax = mplstereonet.subplots()    
        cax = ax.density_contourf(strike, dip, measurement='poles', weights = weight, method='schmidt',gridsize=25,extend='neither',cmap='jet')
        ax.grid()
        ax.set_title('DFN stereonet')
        ax.set_azimuth_ticks([])
        ax.set_xlabel('Dip')
        fig.colorbar(cax)
        imgpath = os.path.join('.','stereonets')
        imgname=os.path.join(imgpath,'stereonet.png')
        if (len(dip)<2000):
            ax.pole(strike, dip, 'k+', markersize=1, markeredgewidth=1)
        if not os.path.isdir(imgpath):
            os.makedirs(imgpath)
        pl.savefig(imgname)
        pl.show()
        pl.close()

def launch_test(seed=1234, size=10, plot=True):
    # parameters
    system_size = size
    system_center = [0,0,0]
    seed = seed
    percol = 10
    # ---------------- initialization of DFN domain ----------------
    print('initialization')
    system = dfnB.System()
    # as system, only the parallelepipedic system has a constructor
    # for other systems, only manual building is available
    system.buildParallelepiped(system_center, system_size)
    # ---------------------- generate the DFN ----------------------
    print('generation')
    dfn = dfnB.DFN(system)
    # Poisson DFN generator
    generator = dfnG.PoissonGenerator(dfn,seed)
    # sizes:
    # setPowerlawSizes 
    # setUniformSizes
    # setExponentialSizes
    # setBootstrappedSizes
    generator.setSizesPowerlaw(4)
    # positions
    # setUniformPositions
    # setBootstrappedPositions
    generator.setPositionsUniform()
    # orientations
    # setUniformOrientations
    # setFisherOrientations
    # setBootstrappedOrientations
    generator.setOrientationsUniform()
    # stop criterion
    # setStopDensity
    # setStopPercolation
    # setStopNumber
    generator.setStopPercolation(percol)
    # launch the generation
    generator.generate()
    # --------------------- connectivity graph ---------------------
    print('connectivity')
    dfn.computeIntersections()    
    # get the backbone for permeameter conditions
    dgraph = dfnGr.DFNGraph(dfn,{0:1,1:1})
    dfn_backbone = dgraph.getDFN(label='backbone')
    dfn_backbone.computeIntersections()
    # ----------------------- basic analyses -----------------------
    print('analyses')
    analyser = dfnA.DFNAnalyser(dfn)
    analyser_backbone = dfnA.DFNAnalyser(dfn_backbone)
    print("nf all={}".format(dfn.nbFractures()))
    print("p32 all={}".format(analyser.p32()))
    print("p all={}".format(analyser.percolation(True)))
    print("nf backbone={}".format(dfn_backbone.nbFractures()))
    print("p32 backbone={}".format(analyser_backbone.p32()))
    print("p backbone={}".format(analyser_backbone.percolation(True)))
    # ---------------------- meshing and flow ----------------------
    mesh = dfn_mesh.DFNMesh(dfn_backbone)
    mesh.manageIntersections(1e-5)
    mesh.generateMesh("bamg")
    # transmissivities
    print("set T")
    #hydro = dfn_hydro.HydraulicProperties(dfn_backbone,dfn_hydro.HydraulicProperties.TransmissivityType.constant,[ 1 ])
    hydro = dfn_hydro.HydraulicProperties(dfn_backbone,dfn_hydro.HydraulicProperties.TransmissivityType.crf,[0.1,int(rng.CRFGenerator.DistributionName.lognormal),0,1,int(rng.CRFGenerator.CorrelationName.self_affine),3,3,1001])
    mesh.setHydraulicProperties(hydro)
    # boundary conditions
    print("set DC")
    D = dfnBC.DFNDomainBoundaryCondition()
    D.setPermeameterCondition(dfn_backbone,10,1)
    mesh.setBoundaryCondition(D)
    # flow solving
    print("solver")
    [p,Tp,Q,U] = dfn_solver.solveDarcyProblem(mesh,D)
    # --------------------------- VISUALIZATION BY VTK ----------------------------------
    Aw = dfnIO.DFNIO_Object(mesh)
    Aw.appendDataOnMesh(p,"pression")
    Aw.appendDataOnMesh(U,"flux")
    Aw.writeVTKOnMesh("Mesh_Current.vtp",True)
    if plot:
        Aw.windowsRender("flux",minval=0.001,islog=True,viewmesh=False)
    # some post-processing
    postTraitment = dfn_solver.DFNSolverPostProcess()
    inoutFluxSystem =  postTraitment.inputOutputFluxSystem(Q,mesh)
    print ('Input, output of the system: \n')
    print(inoutFluxSystem)
    print(abs(inoutFluxSystem[1] +inoutFluxSystem[0]) )
    FluxFractureVector = postTraitment.fluxAllFractures(Q,mesh)   


#----------------------------------------------------
# DFN testing
# ---------------------------------------------------

if __name__ == '__main__':
    launch_test(size=10)

An example with DFN generation and some analyses


import sys, os

import dfnlab.Geometry as dfn_geo
import dfnlab.DFNBasis as dfn

import matplotlib.colors
import numpy as np
import pylab as pylab

def test_analyser(L,lmin,lmax,a,p32):

    # Generate network
    print('Generate network...')
    system = dfn.System()
    system.buildParallelepiped([0, 0, 0],L)
    well = dfn.Well1D([0, 0, 0.5*L], [0, 0, - 0.5*L])
    system.addWellTunnel(well)
    fnet = dfn.DFN(system)
    generator = dfn_gen.PoissonGenerator(fnet, 1001)
    generator.setSizesPowerlaw(a,lmin,lmax)
    generator.setPositionsUniform()
    set1 = dfn_analysis.FractureSetFisher(1, 0, 45, 18)
    set2 = dfn_analysis.FractureSetFisher(2, 90, 45, 18)
    set3 = dfn_analysis.FractureSetFisher(3, 323, 87, 18)
    setList = [set1,set2,set3]
    for fset in setList:
        generator.setOrientationsFisher(fset.getDip(),fset.getDipDir(),fset.getKappa())
        generator.setStopDensity(p32/len(setList))
        generator.generate()
    # Write disk file
    DFNname = r'dfn_a=' + str(a) + '_lmin=' + str(lmin) + '_lmax=' + str(lmax) + '_p32=' + str(p32)
    dfn.write_disk_file(fnet,DFNname + '.disk') # Disk file format

    # Create analyser of the network
    analyser = dfn_analysis.DFNAnalyser(fnet)

    # Identify fracture sets
    analyser.identifyFractureSets(setList)
    oriDict={fset.getId():[] for fset in setList}
    for frac in fnet:
        strike=frac.dipD()[1]-90
        dip=frac.dipD()[0]
        oriDict[frac.getExtra('FractureSet')].append((strike,dip))
    plot_stereonet(oriDict,r'stereonet.png')
    dfn_io.write_DFN_file(fnet, DFNname +'.vtp', ["FractureSet"],dfn_io.DFNFileFormat.VTP,25,True,True) # VTP format


    # Size distribution
    print('Size distribution...')
    nl = analyser.sizeDistribution(20); # number of bins
    dfn_analysis.DFNAnalyser.write_distribution(nl,'size_distribution.dat')
    plot_size_distribution(nl, r'size_distribution.png')

    # Network densities
    print('Analyse network densities...')
    p10 = analyser.p10(0) # well id
    p30 = analyser.p30()
    p32b = analyser.p32()
    percolation = analyser.percolation()
    print("p10 = {}".format(p10))
    print("p30 = {}".format(p30))
    print("p32 = {}".format(p32b))
    print("p = {}".format(percolation))
    dfn_io.write_density_map(fnet, DFNname +'_densityMap.vtk',2.5,True)

    # Correlation pair function of fractures centers
    print('Correlation pair function of fractures centers...')
    c2r_3D = analyser.correlationPairFunction3D(0.5,30) # minimum pair distance / number of bins
    dfn_analysis.DFNAnalyser.write_distribution(c2r_3D,r'correlation_pair_function.dat')
    plot_correlation_pair_function(c2r_3D, r'correlation_pair_function_3D.png')

    # Lacunarity of 3D fracture densities
    print('Lacunarity of 3D fracture densities...')
    lacu3D = analyser.lacunarity3D(2.5,0.) # minimum size / overlap ratio
    dfn_analysis.DFNAnalyser.write_distribution(lacu3D,'lacuanrity3D.dat')
    plot_lacu3D(lacu3D, r'lacunarity3D')


def plot_stereonet(oriDict,figureName):
    fig, ax = mplstereonet.subplots()
    cmap = pyplot.cm.rainbow
    norm = matplotlib.colors.Normalize(vmin=min(list(oriDict.keys())), vmax=max(list(oriDict.keys())))
    for fset,oriset in oriDict.items():
        strikes = [elt[0] for elt in oriset]
        dips = [elt[1] for elt in oriset]
        ax.pole(strikes, dips,markersize=0.2,color=cmap(norm(fset)))
    ax.grid()
    pyplot.savefig(figureName)
    pyplot.show()
    pyplot.close()

def plot_size_distribution(size_distribution, figureName):
    csfont = {'fontname':'Times New Roman'}
    pyplot.rcParams['mathtext.fontset'] = 'dejavuserif'
    fig = pyplot.figure()
    ax = pyplot.axes()
    ax.set_xscale("log")
    ax.set_yscale("log")
    l = np.array(size_distribution['Val'])
    idx1 = [True if l[i]!=0 else False for i in range(0,len(l)) ]
    nl = np.array(size_distribution['Pdf'])
    idx2 = [True if nl[i]!=0 else False for i in range(0,len(nl)) ]
    idx = idx1 and idx2
    l = l[idx]
    nl = nl[idx]
    ax.plot(l,nl,marker='o',markersize=5, linewidth=0, color='k',label='DFN')

    log_l = np.log10(l)
    log_nl = np.log10(nl)
    idx = np.isfinite(log_l) & np.isfinite(log_nl)
    m, b = pylab.polyfit(log_l[idx],log_nl[idx],1)
    a=-m
    alpha = 10**b 
    fit = [alpha*pow(elt,-a) for elt in l]
    lab = '~$'+"{:.2e}".format(alpha)+'*l^{-'+str(round(a,2))+'}$'
    ax.plot(l,fit,marker='o', linestyle='dashed', markersize=0, linewidth=2, color='grey', label=lab)

    pyplot.xlabel('Fracture size $l$',**csfont)
    pyplot.ylabel('Fracture size distribution $n(l)$',**csfont)
    pyplot.legend()
    fig.savefig(figureName)
    pyplot.close(fig)


def plot_correlation_pair_function(c2r, figureName):
    csfont = {'fontname':'Times New Roman'}
    pyplot.rcParams['mathtext.fontset'] = 'dejavuserif'
    x = np.array(c2r['r_mean'])
    y1 = np.array(c2r['correlation_pair_function'])
    y2 = np.array(c2r['correlation_dimention'])
    idx1 = [True if y1[i]!=0 else False for i in range(0,len(y1)) ]
    idx2 = [True if y2[i]!=0 else False for i in range(0,len(y2)) ]
    idx = np.isfinite(y1) & np.isfinite(y2) & idx1 & idx2
    r = x[idx]
    c2r_val = y1[idx]
    D2 = y2[idx]
    csfont = {'fontname':'Times New Roman'}
    fig, ax1 = pyplot.subplots()

    ax1.set_xscale("log")
    ax1.set_yscale("log")
    ax1.plot(r, c2r_val, marker='o',markersize=5, linewidth=2, color='k',label='$C_{2}(r)$')
    ax1.set_xlabel('Pair distance $r$')
    ax1.set_ylabel('Correlation pair function $C_{2}(r)$',color='k')
    ax1.tick_params(axis='y', labelcolor='k')
    ax1.legend()

    ax2 = ax1.twinx() 
    ax2.set_ylabel('Correlation dimension $D_{2}$', color='b')
    ax2.plot(r, D2, marker='o', linestyle='dashed', markersize=0, linewidth=1, color='b',label='$D_{2}(r)$')
    ax2.tick_params(axis='y', labelcolor='b')
    ax2.legend()

    fig.savefig(figureName)
    pyplot.close(fig)

dfnlab.DFNBasis Module¶

DFNBasis python For DFN manipulations. These module contains all relevant classes for creating DFN related objects:

  • geological system
  • discrete fractures

  • simple input/output

This is the elementary module of DFNLab, it should be always included.
